
# Create multiple Ec2 instances in different AZ
# SSH into EC2 instance
sudo yum install -y amazon-efs-utils  [Install EFS mount helper]
sudo mkdir /efs [create new directory 'efs']
#edit security group as type:NFS, source: sg for EFS
sudo mount -t efs 'file system ID':/ /efs [mount file system using EFS mount helper (this method is without encryption)]
# To verify:
cd /efs
sudo su
echo "hello" > hello.txt
cat hello.txt
# use command 'ls' and/or 'cat hello.txt' on different instance, if the file is available, mount is successful
